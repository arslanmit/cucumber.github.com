def file_and_line
  path, line = *@proc.inspect.match(PROC_PATTERN)[1..2]
  path = File.expand_path(path)
  pwd = File.expand_path(PWD)
  if path.index(pwd)
    path = path[pwd.length+1..-1]
  elsif path =~ /.*\/gems\/(.*\.rb)$/
    path = $1
  end
  [path, line.to_i]
end

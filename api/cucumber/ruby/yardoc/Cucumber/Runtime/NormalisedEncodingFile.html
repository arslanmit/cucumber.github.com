def initialize(path)
  begin
    @file = File.new(path)
    set_encoding
  rescue Errno::EACCES => e
    raise FileNotFoundException.new(e, File.expand_path(path))
  rescue Errno::ENOENT => e
    raise FeatureFolderNotFoundException.new(e, path)
  end
end

def accept(visitor)
  return if Cucumber.wants_to_quit
  @steps.each do |step|
    visitor.visit_step(step)
  end
end

def step_invocations(background = false)
  StepCollection.new(@steps.map{ |step|
    i = step.step_invocation
    i.background = background
    i
  })
end
